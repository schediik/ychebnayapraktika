def caesar_cipher(text, shift):
    result = []
    for char in text:
        if char.isalpha():
            start = ord('А') if char.isupper() else ord('а')
            offset = ((ord(char) - start + shift) % 26)
            new_char = chr(start + offset)
        else:
            new_char = char
        result.append(new_char)
    return ''.join(result)
